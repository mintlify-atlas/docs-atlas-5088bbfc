## List Events

<ParamField query="GET" path="/api/v0/events/">
  Retrieve a paginated list of OWASP events.
</ParamField>

### Query Parameters

<ParamField query="is_upcoming" type="boolean" optional>
  Filter for upcoming events only.
  - `true` - Only return events that haven't ended yet
  - `false` or omit - Return all events
</ParamField>

<ParamField query="latitude_gte" type="float" optional>
  Filter events with latitude greater than or equal to the specified value.
  
  Example: `40.0` (events at or above 40° North)
</ParamField>

<ParamField query="latitude_lte" type="float" optional>
  Filter events with latitude less than or equal to the specified value.
  
  Example: `50.0` (events at or below 50° North)
</ParamField>

<ParamField query="longitude_gte" type="float" optional>
  Filter events with longitude greater than or equal to the specified value.
  
  Example: `-10.0`
</ParamField>

<ParamField query="longitude_lte" type="float" optional>
  Filter events with longitude less than or equal to the specified value.
  
  Example: `30.0`
</ParamField>

<ParamField query="ordering" type="string" optional>
  Sort results by field. Prefix with `-` for descending order.
  - `start_date` - Event start date (ascending)
  - `-start_date` - Event start date (descending)
  - `end_date` - Event end date (ascending)
  - `-end_date` - Event end date (descending)
  - `latitude` - Latitude (ascending)
  - `-latitude` - Latitude (descending)
  - `longitude` - Longitude (ascending)
  - `-longitude` - Longitude (descending)
  
  Default: `-start_date,-end_date` (for all events), `start_date,end_date` (for upcoming events)
</ParamField>

<ParamField query="page" type="integer" optional>
  Page number for pagination. Default: `1`
</ParamField>

<ParamField query="limit" type="integer" optional>
  Number of results per page. Default: `100`
</ParamField>

### Response Schema

<ResponseField name="items" type="array">
  Array of event objects
  
  <Expandable title="Event Object">
    <ResponseField name="end_date" type="string" optional>
      Event end date in ISO 8601 format (YYYY-MM-DD)
    </ResponseField>
    
    <ResponseField name="key" type="string">
      Unique event identifier
    </ResponseField>
    
    <ResponseField name="latitude" type="float" optional>
      Geographic latitude of the event location
    </ResponseField>
    
    <ResponseField name="longitude" type="float" optional>
      Geographic longitude of the event location
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Event name
    </ResponseField>
    
    <ResponseField name="start_date" type="string">
      Event start date in ISO 8601 format (YYYY-MM-DD)
    </ResponseField>
    
    <ResponseField name="url" type="string" optional>
      Event URL for more information
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="count" type="integer">
  Total number of events matching the filter
</ResponseField>

### Examples

<Tabs>
  <Tab title="cURL">
    ```bash
    # List all events
    curl -X GET "https://nest.owasp.org/api/v0/events/"
    
    # List upcoming events only
    curl -X GET "https://nest.owasp.org/api/v0/events/?is_upcoming=true"
    
    # Filter by geographic bounds (North American region)
    curl -X GET "https://nest.owasp.org/api/v0/events/?latitude_gte=25&latitude_lte=50&longitude_gte=-125&longitude_lte=-65"
    
    # Sort by start date (soonest first)
    curl -X GET "https://nest.owasp.org/api/v0/events/?ordering=start_date"
    
    # Get upcoming events with pagination
    curl -X GET "https://nest.owasp.org/api/v0/events/?is_upcoming=true&page=1&limit=20"
    ```
  </Tab>
  
  <Tab title="Python">
    ```python
    import requests
    
    # List all events
    response = requests.get("https://nest.owasp.org/api/v0/events/")
    events = response.json()
    
    # List upcoming events only
    response = requests.get(
        "https://nest.owasp.org/api/v0/events/",
        params={"is_upcoming": True}
    )
    
    # Filter by geographic bounds (North American region)
    response = requests.get(
        "https://nest.owasp.org/api/v0/events/",
        params={
            "latitude_gte": 25,
            "latitude_lte": 50,
            "longitude_gte": -125,
            "longitude_lte": -65
        }
    )
    
    # Sort by start date (soonest first)
    response = requests.get(
        "https://nest.owasp.org/api/v0/events/",
        params={"ordering": "start_date"}
    )
    
    # Print upcoming events
    upcoming = requests.get(
        "https://nest.owasp.org/api/v0/events/",
        params={"is_upcoming": True}
    ).json()
    
    for event in upcoming['items']:
        print(f"{event['name']} - {event['start_date']}")
    ```
  </Tab>
  
  <Tab title="JavaScript">

    ```javascript
    geoUrl.searchParams.set('longitude_gte', '-125');
    geoUrl.searchParams.set('longitude_lte', '-65');
    const geoResponse = await fetch(geoUrl);
    
    // Sort by start date (soonest first)
    const sortedUrl = new URL('https://nest.owasp.org/api/v0/events/');
    sortedUrl.searchParams.set('ordering', 'start_date');
    const sortedResponse = await fetch(sortedUrl);
    
    // Print upcoming events
    const upcoming = await upcomingResponse.json();
    upcoming.items.forEach(event => {
      console.log(`${event.name} - ${event.start_date}`);
    });
    ```

## Get Event

<ParamField query="GET" path="/api/v0/events/{event_id}">
  Retrieve detailed information about a specific OWASP event.
</ParamField>

### Path Parameters

<ParamField path="event_id" type="string" required>
  Event identifier (key).
  
  Example: `owasp-global-appsec-usa-2025-washington-dc`
</ParamField>

### Response Schema

<ResponseField name="end_date" type="string" optional>
  Event end date in ISO 8601 format (YYYY-MM-DD)
</ResponseField>

<ResponseField name="key" type="string">
  Unique event identifier
</ResponseField>

<ResponseField name="latitude" type="float" optional>
  Geographic latitude of the event location
</ResponseField>

<ResponseField name="longitude" type="float" optional>
  Geographic longitude of the event location
</ResponseField>

<ResponseField name="name" type="string">
  Event name
</ResponseField>

<ResponseField name="start_date" type="string">
  Event start date in ISO 8601 format (YYYY-MM-DD)
</ResponseField>

<ResponseField name="url" type="string" optional>
  Event URL for more information
</ResponseField>

<ResponseField name="description" type="string" optional>
  Event description
</ResponseField>

### Error Responses

<ResponseField name="404 Not Found" type="object">
  <Expandable title="Error Schema">
    <ResponseField name="message" type="string">
      Error message: "Event not found"
    </ResponseField>
  </Expandable>
</ResponseField>

### Examples

<Tabs>
  <Tab title="cURL">
    ```bash
    # Get event by key
    curl -X GET "https://nest.owasp.org/api/v0/events/owasp-global-appsec-usa-2025-washington-dc"
    ```
  </Tab>
  
  <Tab title="Python">
    ```python
    import requests
    
    # Get event details
    event_id = "owasp-global-appsec-usa-2025-washington-dc"
    response = requests.get(f"https://nest.owasp.org/api/v0/events/{event_id}")
    
    if response.status_code == 200:
        event = response.json()
        print(f"Name: {event['name']}")
        print(f"Dates: {event['start_date']} to {event['end_date']}")
        print(f"Location: {event.get('latitude')}, {event.get('longitude')}")
        print(f"URL: {event.get('url')}")
    elif response.status_code == 404:
        print("Event not found")
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```javascript
    // Get event details
    const eventId = 'owasp-global-appsec-usa-2025-washington-dc';
    const response = await fetch(`https://nest.owasp.org/api/v0/events/${eventId}`);
    
    if (response.ok) {
      const event = await response.json();
      console.log(`Name: ${event.name}`);
      console.log(`Dates: ${event.start_date} to ${event.end_date}`);
      console.log(`Location: ${event.latitude}, ${event.longitude}`);
      console.log(`URL: ${event.url}`);
    } else if (response.status === 404) {
      console.log('Event not found');
    }
    ```
  </Tab>
</Tabs>
